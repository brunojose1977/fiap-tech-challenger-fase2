331 from reportlab.lib.pagesizes import A4 

332 from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer 

333 from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle 

334 from reportlab.lib.enums import TA_CENTER

337 analise_texto = """ 

338# Título Principal 

339 

340## Subtítulo 

341 

342### Sub-subtítulo 

343 

344Texto normal com **negrito** e *itálico*. 

345 

346Mais texto aqui. 

347""" 

348 

349 def gerar_relatorio_pdf(analise_texto, output_path, _dados_ranking=None): 

350 if not analise_texto: 

351 return False 

352 

353 try: 

354 doc = SimpleDocTemplate(output_path, pagesize=A4) 

355 story = [] 

356 styles = getSampleStyleSheet() 

357 

358 # Adiciona conteúdo 

359 title_style = ParagraphStyle( 

360 'CustomTitle', 

361 parent=styles['Heading1'], 

362 fontSize=20, 

363 alignment=TA_CENTER 

364 ) 

365 

366 story.append(Paragraph("Relatório de Análise de Resultados", title_style)) 

367 story.append(Spacer(1, 0.2)) 

368 

369 # Processa texto 

370 linhas = analise_texto.split('\n') 

371 for linha in linhas: 

372 linha = linha.strip() 

373 if not linha: 

374 continue 

375 

376 if linha.startswith('###'): 

377 titulo = linha.replace('###', '').strip() 

378 if titulo: 

379 story.append(Paragraph(f"<b>{titulo}</b>", styles['Heading3'])) 

380 elif linha.startswith('##'): 

381 titulo = linha.replace('##', '').strip() 

382 if titulo: 

383 story.append(Paragraph(f"<b>{titulo}</b>", styles['Heading2'])) 

384 elif linha.startswith('#'): 

385 titulo = linha.replace('#', '').strip() 

386 if titulo: 

387 story.append(Paragraph(f"<b>{titulo}</b>", styles['Heading1'])) 

388 else: 

389 texto_formatado = linha 

390 texto_formatado = re.sub(r'\*\*(.+?)\*\*', r'<b>\1</b>', texto_formatado) 

391 texto_formatado = re.sub(r'(?<!\*)\*([^*]+?)\*(?!\*)', r'<i>\1</i>', texto_formatado) 

392 if texto_formatado.strip(): 

393 story.append(Paragraph(texto_formatado, styles['Normal'])) 

394 

395 doc.build(story) 

396 return True 

397 

398 except (IOError, OSError) as e: 

399 print(f"Erro ao gerar PDF: {str(e)}") 

400 return False 

401 

402 # Testa geração de PDF 

403 with tempfile.NamedTemporaryFile(mode='w', suffix='.pdf', delete=False) as tmp_file: 

404 output_path = tmp_file.name 

405 

406 try: 

407 result = gerar_relatorio_pdf(analise_texto, output_path) 

408 assert result is True, "PDF deveria ser gerado com sucesso" 

409 assert os.path.exists(output_path), "Arquivo PDF deveria existir" 

410 assert os.path.getsize(output_path) > 0, "Arquivo PDF não deveria estar vazio" 

411 

412 finally: 

413 if os.path.exists(output_path): 

414 os.unlink(output_path)
